97 std::optional<edge_type> find(std::ranges::forward_range auto&& str) const
98 {
99 if (nodes_.empty())
100 return std::nullopt;
101 auto first = std::ranges::begin(str);
102 auto last = std::ranges::end(str);
103 for (auto edge = edge_type{};;) {
104 auto lbl = label(edge.child_node);
105 edge.labels_len += lbl.length();
106 auto diff = std::mismatch(
107 first, last, data() + lbl.first, data() + lbl.last, eq_);
108 if (diff.first == last)
109 return edge;
110 if (diff.second != data() + lbl.last || leaf(edge.child_node))
111 return std::nullopt;
112 auto it = nodes_[edge.child_node].children.find(*diff.first);
113 if (it == nodes_[edge.child_node].children.end())
114 return std::nullopt;
115 first = diff.first;
116 edge.parent_node = std::exchange(edge.child_node, it->second);
117 }

122 generator<edge_type> depth_first_search(const edge_type& start) const
123 {
124 for (auto stack = std::stack<edge_type>{{start}}; !stack.empty();) {
125 auto edge = stack.top();
126 co_yield edge;
127 stack.pop();
128 if (!leaf(edge.child_node))
129 std::ranges::copy(
130 try_reverse(nodes_[edge.child_node].children) |
131 std::views::transform([&](auto&& item) {
132 return edge_type{
133 edge.child_node,
134 item.second,
135 edge.labels_len + label(item.second).length()};
136 }),
137 insert_iterator(stack));
138 }

163 bool split(Size& node)
164 {
165 auto lbl = label(node);
166 Size cut = lbl.first + (size() - pos_) - 1;
167 Size back = size() - 1;
168 if (eq_(str_[cut], str_[back]))
169 return false;
170 Size old = std::exchange(node, (Size)nodes_.size());

174 if (!leaf(old))
175 nodes_[old].label = {cut, lbl.last};
176 return true;
177 }
